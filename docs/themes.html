How to Use

<!-- Load a theme CSS file -->
<link rel="stylesheet" href="velinstyle/dist/themes/brutalist.min.css">

<!-- Or scope to a section -->
<div data-velin-theme="soft">
  <button class="velin-btn velin-btn--primary">Soft Button</button>
</div>

<!-- Dark mode -->
<html data-velin-theme="dark">
